深度封装typeof判断

function myTypeof(val) {
    var type = typeof(val)
    var res = {
        '[object Object]' : 'object',
        '[object Array]' : 'array',
        '[object Number]' : 'object number',
        '[object String]' : 'object string',
        '[object Boolean]' : 'object boolean'
    }
    if (val === null) {
        return 'null'
    } else if (type == 'object') {
        var str = Object.prototype.toString.call(val)
        return res[str]
    } else {
        return type
    }
}
